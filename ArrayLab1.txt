#include <stdio.h>
int main() {
   int n,sum=0;
   printf("Input no.of elements:\n");
   scanf("%d",&n);
   int ar[n];
   printf("Enter elements:\n");
   for(int i=0;i<n;i++)
     scanf("%d",&ar[i]);
   for(int i=0;i<n;i++)
     sum=sum+ar[i];
     printf("Sum of elements: %d",sum);
    return 0;
}
